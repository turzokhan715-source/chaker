const baseApiUrl = "https://email-reader-api--turzokhan715.replit.app/api/email/search"; async function getFacebookCode() { const rawData = document.getElementById('mail-input').value; if (!rawData || !rawData.includes('|')) { alert("Please enter input in 'email|password' format!"); return; } const parts = rawData.split('|'); const userEmail = parts[0].trim(); const userPass = parts[1].trim(); const finalUrl = `${baseApiUrl}?email=${encodeURIComponent(userEmail)}&password=${encodeURIComponent(userPass)}&query=Facebook`; try { document.getElementById('code-display').innerText = "Searching code, please wait..."; const response = await fetch(finalUrl); const data = await response.json(); if (data && data.length > 0) { const latestEmail = data[0]; const emailText = latestEmail.body || latestEmail.subject || ""; const codeMatch = emailText.match(/\d{6}/); if (codeMatch) { document.getElementById('code-display').innerText = codeMatch[0]; } else { document.getElementById('code-display').innerText = "Code not found in email!"; } } else { document.getElementById('code-display').innerText = "No Facebook emails found!"; } } catch (error) { console.error("Error:", error); document.getElementById('code-display').innerText = "Connection failed or Login incorrect!"; } }
